// 1. Two Sum
impl Solution {
    pub fn two_sum(nums: Vec<i32>, target: i32) -> Vec<i32> {
        let mut hash = std::collections::HashMap::with_capacity(nums.len());

        for (i, &n) in nums.iter().enumerate() {
            let diff = target - n;
            if let Some(&j) = hash.get(&diff) {
                return vec![i as i32, j as i32];
            }
            hash.insert(n, i);
        }

        vec![]
    }
}
