// 1411. Number of Ways to Paint N Ã— 3 Grid
impl Solution {
    pub fn num_of_ways(n: i32) -> i32 {
        const MOD: i64 = 1_000_000_007;
        let mut a1 = 6;
        let mut a2 = 6;
        let mut b;
        for  i in 1..n {
            b = a1*2 + a2*2;
            a1 = (b + a1) % MOD;
            a2 = b % MOD;
        }
        return ((a1 + a2) % MOD) as i32;
    }
}
